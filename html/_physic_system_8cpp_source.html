R-Type 1.0
R-Type is a 3rd year Epitech project where a group of 4-6 students have to code an ersatz of the original horizontally scrolling shooter arcade video game from 1987 of the same name. In the span of 2 months, the goal is to learn the ECS architectural pattern to code the game in C++ with an additional 2-4 players coop.

106 std::size_t physicApp = (InfoComp::APP);

108 std::size_t physicEne = (InfoComp::ENEMY);
109 std::size_t physicDrop = (InfoComp::DROP);
110
111 if (masks[i].has_value() && (masks[i].value() & physicProj) == physicProj) {
112 Parent &par = componentManager.getSingleComponent<Parent>(i);
113 for (std::size_t j = 0; j < masks.size(); j++) {
114 if (masks[j].has_value() && ((masks[j].value() & physicCon) == physicCon || (masks[j].value() & physicEne) == physicEne) && par.id != j) {
115 if ((masks[j].value() & physicApp) == physicApp && componentManager.getSingleComponent<Appearance>(j).app)
116 continue;
117 if (this->checkColision(pos, componentManager.getSingleComponent<Position>(j), componentManager.getSingleComponent<Size>(i),
118 componentManager.getSingleComponent<Size>(j))) {
119 Life &hp = componentManager.getSingleComponent<Life>(j);
120 Projectile &proj = componentManager.getSingleComponent<Projectile>(i);
121 if (proj.damage >= hp.life) {
122 if ((masks[j].value() & physicDrop) == physicDrop)
123 this->createBonus(j, componentManager.getSingleComponent<DropBonus>(j).id, componentManager, entityManager);
124 hp.life = 0;
125 componentManager.removeAllComponents(j);
126 entityManager.removeMask(j);
127 } else
128 hp.life -= proj.damage;
129 componentManager.removeAllComponents(i);
130 entityManager.removeMask(i);
131 return true;
132 }

139void PhysicSystem::update(ComponentManager &componentManager, EntityManager &entityManager)
140{
141 auto &masks = entityManager.getMasks();
142 std::size_t physicSpeed = (InfoComp::VEL | InfoComp::POS);
143 std::size_t physicControl = (InfoComp::CONTROLLABLE | InfoComp::POS);
144 std::size_t physicPar = (InfoComp::VEL | InfoComp::POS | InfoComp::PARALLAX);
145 std::size_t physicPat = (InfoComp::PATERN);
146 std::size_t physicAppear = (InfoComp::APP);
147
148 for (std::size_t i = 0; i < masks.size(); i++) {
149 if (masks[i].has_value() && (masks[i].value() & physicSpeed) == physicSpeed) {
150 Position &pos = componentManager.getSingleComponent<Position>(i);
151 Velocity &vel = componentManager.getSingleComponent<Velocity>(i);
152 if ((masks[i].value() & physicAppear) == physicAppear && checkAppareance(componentManager, i, pos, vel))
153 continue;
154 if ((masks[i].value() & physicPar) == physicPar) {
155 pos.x += vel.x;
156 if (pos.x <= -static_cast<int>(_window->getSize().x))
157 pos.x = 0;
158 continue;
159 }
160 if (pos.x > _window->getSize().x || pos.y > _window->getSize().y || pos.x < -100 || pos.y < -100) {
161 entityManager.removeMask(i);
162 componentManager.removeAllComponents(i);
163 continue;
164 }
165 if ((masks[i].value() & physicPat) != physicPat) {
166 pos.x += vel.x;
167 pos.y += vel.y;
168 if (this->collisionFireball(i, componentManager, entityManager, pos))
169 continue;
170 if (this->collisionEnemy(i, componentManager, entityManager, pos))
171 continue;
172 if (this->collisionBonus(i, componentManager, entityManager, pos))
173 continue;
174 if ((masks[i].value() & physicControl) == physicControl) {
175 Position &pos = componentManager.getSingleComponent<Position>(i);
176 pos.x < 0 ? pos.x = 0 : pos.x;
177 pos.y < 0 ? pos.y = 0 : pos.y;
178 pos.x > _window->getSize().x - 100 ? pos.x = _window->getSize().x - 100 : pos.x;
179 pos.y > _window->getSize().y - 100 ? pos.y = _window->getSize().y - 100 : pos.y;
180 continue;
181 }
182 }
183 }
